bool resolver_skyscraper(int tabuleiro[N][N], int dicas[16]) {
    int linha, col;

    // 1. Se não houver mais células vazias, o quebra-cabeça está resolvido!
    if (!proxima_celula_vazia(tabuleiro, &linha, &col)) {
        return true; 
    }

    // 2. Tentar números de 1 a N
    for (int num = 1; num <= N; num++) {
        // 3. Verificar se 'num' é seguro naquela posição
        if (eh_valido(tabuleiro, linha, col, num, dicas)) {
            tabuleiro[linha][col] = num;

            // 4. Ir para o próximo passo
            if (resolver_skyscraper(tabuleiro, dicas)) {
                return true;
            }

            // 5. Se falhou lá na frente, desfazemos a escolha (Backtracking)
            tabuleiro[linha][col] = 0;
        }
    }
    return false; // Aciona o backtrack no nível anterior
}
